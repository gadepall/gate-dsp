\begin{lemma}{Linearity of Laplace Transform}
\begin{align}
    \mathcal{L}\cbrak{af\brak{t}+bg\brak{t}}=a\mathcal{L}\cbrak{f\brak{t}}+b\mathcal{L}\cbrak{g\brak{t}}
\end{align}
\end{lemma}

\begin{align}
    s^{2}Y\brak{s}+4sY\brak{s}+3Y\brak{s}&=2sX\brak{s}+4X\brak{s}\\

\end{align}
